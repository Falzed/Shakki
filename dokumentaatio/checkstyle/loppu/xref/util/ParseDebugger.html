18          Scanner lukija = new Scanner(System.in);
19          Lauta lauta = new Lauta();
20          lauta.alustaLauta();
21          int[] koordinaatit = new int[2];
22          koordinaatit[0] = 4;
23          koordinaatit[1] = 4;
24          Torni torni = new Torni("valkoinen");
25          Torni torni2 = new Torni("valkoinen");
26          LaudanMuutokset.aseta(torni, koordinaatit, lauta);
27          koordinaatit[0] = 4;
28          koordinaatit[1] = 7;
29          LaudanMuutokset.aseta(torni2, koordinaatit, lauta);
30          Nappula.Puoli vuoro = Nappula.Puoli.VALKOINEN;
31          while (true) {
32              line = lukija.nextLine();
33              if (line.isEmpty()) {
34                  break;
35              }
36              if (line.equals("algebraic")) {
37                  line = lukija.nextLine();
38                  int[] table = logic.parser.Parser.parseAlgebraic(line);
39                  System.out.println(table[0] + "," + table[1]);
40              }
41              if (line.equals("musta")) {
42                  vuoro = Nappula.Puoli.MUSTA;
43              }
44              if (line.equals("valkoinen")) {
45                  vuoro = Nappula.Puoli.VALKOINEN;
46              }
47              ParserReturn parserReturn = logic.parser.Parser.parseCommand(line, vuoro, lauta);
48              
49              System.out.println(parserReturn);
50  
51              if (parserReturn.getCoordinates() == null || parserReturn.getCoordinates()[0] == null) {
52                  System.out.println("table null");
53                  System.out.println("table.length " + parserReturn.getCoordinates().length);
54                  System.out.println("table[0].length " + parserReturn.getCoordinates()[0].length);
55              } else {
56                  System.out.println(parserReturn.getCoordinates()[0][0] + "," + parserReturn.getCoordinates()[0][1] + "-" + parserReturn.getCoordinates()[1][0] + "," + parserReturn.getCoordinates()[1][1]);
57              }
58          }
59      }
