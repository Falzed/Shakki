253     public void updateUI() {
254         historiaKentta.setText(peli.getTurnHistory().toString());
255         for (int i = 0; i < peli.getLauta().getLeveys(); i++) {
256             for (int j = 0; j < peli.getLauta().getPituus(); j++) {
257                 int[] koordinaatit = {i, j};
258                 labels[i][j].setText(Character.toString(peli.getLauta().getNappula(koordinaatit).getMerkki()));
259             }
260         }
261     }

309     public void suoritaKomento(String komento) {
310         ParserReturn komentoReturn = peli.suoritaKomento(komento);
311         int[][] startEndPoints = komentoReturn.getCoordinates();
312 //                System.out.println(komentoReturn);
313         if (!komentoReturn.getError().isEmpty()) {
314             popUpErrorMessage(komentoReturn.getError());
315         }
316         if (startEndPoints != null) {
317             if (peli.tarkistaKorotus(startEndPoints[1])) {
318                 peli.korota(startEndPoints[1], popupKorotus());
319             }
320             historiaKentta.setText(peli.getTurnHistory().toString());
321         }
322 
323         komentoKentta.setText("");
324 
325         updateUI();
326 
327         if (peli.tarkistaMatti()) {
328             popUpErrorMessage(peli.getVuoro() + " on matissa");
329         }
330         if (peli.tarkistaPatti()) {
331             popUpErrorMessage(peli.getVuoro() + " on patissa, tasapeli");
332         }
333     }

341     public void toggleValittu(int[] koordinaatit) {
342         if (panels[koordinaatit[0]][koordinaatit[1]].getBackground().equals(new java.awt.Color(255, 255, 255))) {
343             panels[koordinaatit[0]][koordinaatit[1]].setBackground(new java.awt.Color(200, 200, 200));
344         } else if (panels[koordinaatit[0]][koordinaatit[1]].getBackground().equals(new java.awt.Color(210, 140, 70))) {
345             panels[koordinaatit[0]][koordinaatit[1]].setBackground(new java.awt.Color(160, 90, 20));
346         } else if (panels[koordinaatit[0]][koordinaatit[1]].getBackground().equals(new java.awt.Color(200, 200, 200))) {
347             panels[koordinaatit[0]][koordinaatit[1]].setBackground(new java.awt.Color(255, 255, 255));
348         } else if (panels[koordinaatit[0]][koordinaatit[1]].getBackground().equals(new java.awt.Color(160, 90, 20))) {
349             panels[koordinaatit[0]][koordinaatit[1]].setBackground(new java.awt.Color(210, 140, 70));
350         }
351     }

353     public void unselectAll() {
354         for (int i = 0; i < panels[0].length; i++) {
355             for (int j = 0; j < panels.length; j++) {
356                 if (panels[i][j].getBackground().equals(new java.awt.Color(200, 200, 200))) {
357                     panels[i][j].setBackground(new java.awt.Color(255, 255, 255));
358                 } else if (panels[i][j].getBackground().equals(new java.awt.Color(160, 90, 20))) {
359                     panels[i][j].setBackground(new java.awt.Color(210, 140, 70));
360                 }
361             }
362         }
