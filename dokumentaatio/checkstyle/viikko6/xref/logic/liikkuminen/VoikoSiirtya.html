7   public class VoikoSiirtya {
8       /**
9        *
10       * Metodi tarkistaa, onko annetussa ruudussa olevan nappulan annettuun
11       * kohderuutuun siirtäminen laillinen siirto.
12       *
13       * @param mista lähtöruudun koordinaatit
14       * @param minne kohderuudun koordinaatit
15       * @param lauta lauta jolla nappula ja ruudut ovat
16       * @param enPassant ruutu, josta voi ohesta lyöda sotilaan (jos oma sotilas
17       * oikeassa paikassa)
18       * @return onko siirto laillinen
19       */
20      public static boolean voikoSiirtya(int[] mista, int[] minne, Lauta lauta, int[] enPassant) {
21          if (lauta.getNappula(mista).isEmpty()) {
22              return false;
23          }
24          Nappula nappula = lauta.getNappula(mista);
25          boolean voiko = false;
26          if (nappula.getMerkki() == '\u2655' || nappula.getMerkki() == '\u265B') {
27              voiko = UpseerienLiikkuminen.kuningatarVoikoLiikkua(mista, minne, lauta, nappula.getPuoli());
28          }
29          if (nappula.getMerkki() == '\u2654' || nappula.getMerkki() == '\u265A') {
30              voiko = UpseerienLiikkuminen.kuningasVoikoLiikkua(mista, minne, lauta, nappula.getPuoli());
31          }
32          if (nappula.getMerkki() == '\u2656' || nappula.getMerkki() == '\u265C') {
33              voiko = UpseerienLiikkuminen.torniVoikoLiikkua(mista, minne, lauta, nappula.getPuoli());
34          }
35          if (nappula.getMerkki() == '\u2658' || nappula.getMerkki() == '\u265E') {
36              voiko = UpseerienLiikkuminen.ratsuVoikoLiikkua(mista, minne, lauta, nappula.getPuoli());
37          }
38          if (nappula.getMerkki() == '\u2657' || nappula.getMerkki() == '\u265D') {
39              voiko = UpseerienLiikkuminen.lahettiVoikoLiikkua(mista, minne, lauta, nappula.getPuoli());
40          }
41          if (nappula.getMerkki() == '\u2659' || nappula.getMerkki() == '\u265F') {
42              voiko = SotilasLiikkuminen.sotilasVoikoLiikkua(mista, minne, lauta, nappula.getPuoli(), enPassant);
43          }
44  
45          return voiko;
46      }
47  
48      /**
49       * Metodi tarkistaa, voiko siirron toteuttaa ilman tornitusta.
50       *
51       * @param mista siirrettävän nappulan sijainti
52       * @param minne nappula siirretään
53       * @param lauta lauta jolla liikutaan
54       * @param enPassant ruutu, josta voidaan ohestalyödä sotilas
55       * @return voiko siirron toteuttaa ilman tornitusta
56       */
57      public static boolean voikoSiirtyaTornittamatta(int[] mista, int[] minne, Lauta lauta, int[] enPassant) {
58          Nappula nappula = lauta.getNappula(mista);
59          if (nappula.isEmpty()) {
60              return false;
61          }
62          if (!nappula.getNimi().equals("Kuningas")) {
63              return voikoSiirtya(mista, minne, lauta, enPassant);
64          }
65          return UpseerienLiikkuminen.kuningasVoikoLiikkuaTornittamatta(mista, minne, lauta, nappula.getPuoli());
66      }
67  
68      /**
69       *
70       * Metodi tarkistaa, onko annetussa ruudussa olevan nappulan annettuun
71       * kohderuutuun siirtäminen laillinen siirto.
72       *
73       * @param nappula siirrettävä nappula
74       * @param minne kohderuudun koordinaatit
75       * @param lauta lauta jolla nappula ja ruudut ovat
76       * @param enPassant ruutu, josta voi ohesta lyöda sotilaan (jos oma sotilas
77       * oikeassa paikassa)
78       * @return onko siirto laillinen
79       */
80      public static boolean voikoSiirtya(Nappula nappula, int[] minne, Lauta lauta, int[] enPassant) {
81          return voikoSiirtya(nappula.getKoordinaatit(), minne, lauta, enPassant);
82      }
83  }
