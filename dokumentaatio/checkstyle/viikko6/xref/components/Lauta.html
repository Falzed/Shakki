1   package components;
2   
3   import logic.parser.Parser;
4   
5   /**
6    * Luokka toteuttaa shakkilaudan. Lauta tiet채채 mik채 nappula on miss채kin
7    * ruudussa.
8    *
9    * @author Oskari
10   */
11  public class Lauta {
12  
13      private final int leveys;
14      private final int pituus;
15      private Nappula[][] nappulat;
16  
17      /**
18       * Konstruktori ilman parametreja, olettaa laudan kooksi 8x8.
19       */
20      public Lauta() {
21          this.leveys = 8;
22          this.pituus = 8;
23          nappulat = new Nappula[leveys][pituus];
24      }
25  
26      /**
27       * Kostruktori jolle annettu laudan leveys ja pituus.
28       *
29       * @param leveys laudan leveys
30       * @param pituus laudan pituus
31       */
32      public Lauta(int leveys, int pituus) {
33          this.leveys = leveys;
34          this.pituus = pituus;
35          nappulat = new Nappula[leveys][pituus];
36      }
37  
38      /**
39       * getteri.
40       *
41       * @param koordinaatit koordinaatit
42       * @return nappula
43       */
44      public Nappula getNappula(int[] koordinaatit) {
45          return nappulat[koordinaatit[0]][koordinaatit[1]];
46      }
47  
48      /**
49       * getteri.
50       *
51       * @param string koordinaatit
52       * @return nappula
53       */
54      public Nappula getNappula(String string) {
55          int[] koordinaatit = Parser.parseAlgebraic(string);
56          return getNappula(koordinaatit);
57      }
58  
59      /**
60       * Kopioi laudan.
61       *
62       * @return palauttaa laudan kopion
63       */
64      public Lauta kopioi() {
65          Lauta kopio = new Lauta(leveys, pituus);
66          kopio.alustaLauta();
67          for (int i = 0; i < leveys; i++) {
68              for (int j = 0; j < pituus; j++) {
69                  int[] koordinaatit = {i, j};
70                  Nappula kopioituNappula = getNappula(koordinaatit).kopioi();
71                  logic.LaudanMuutokset.aseta(kopioituNappula, koordinaatit, kopio);
72              }
73          }
74          return kopio;
75      }
76  
77      /**
78       * Alustaa laudan jokaiseen ruutuun Tyhja-olion.
79       */
80      public void alustaLauta() {
81          Tyhja tyhja = new Tyhja();
82          for (int i = 0; i < leveys; i++) {
83              for (int j = 0; j < pituus; j++) {
84                  nappulat[i][j] = tyhja;
85              }
86          }
87      }
88      /**
89       * setteri.
90       * @param nappula nappula joka laitetaan laudalle
91       * @param koordinaatit minne nappula laitetaan
92       */
93      public void setNappula(Nappula nappula, int[] koordinaatit) {
94          nappulat[koordinaatit[0]][koordinaatit[1]] = nappula;
95      }
96  
97      public int getLeveys() {
98          return leveys;
99      }
100 
101     public int getPituus() {
102         return pituus;
103     }
104 }
