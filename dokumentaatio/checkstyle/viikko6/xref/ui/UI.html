229      * Metodi päivittää UI:n.
230      */
231     public void updateUI() {
232         historiaKentta.setText(peli.getTurnHistory().toString());
233         for (int i = 0; i < peli.getLauta().getLeveys(); i++) {
234             for (int j = 0; j < peli.getLauta().getPituus(); j++) {
235                 int[] koordinaatit = {i, j};
236                 labels[i][j].setText(Character.toString(peli.getLauta().getNappula(koordinaatit).getMerkki()));
237             }
238         }
239     }

241     public String popupKorotus() {
242         JFrame frame = new JFrame("Korotus");
243 
244         Object[] possibilities = {"Kuningatar", "Ratsu", "Torni", "Lähetti"};
245         String s = (String) JOptionPane.showInputDialog(
246                 frame,
247                 "",
248                 "Sotilaan korotus",
249                 JOptionPane.PLAIN_MESSAGE,
250                 null,
251                 possibilities,
252                 "Kuningatar");
253         return s;
254     }
