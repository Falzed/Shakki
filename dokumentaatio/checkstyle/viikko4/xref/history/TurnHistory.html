1   package history;

3   import java.util.LinkedList;
4   import variants.*;
5   
6   /**
7    * Luokka toteuttaa laudan vuorohistorian.
8    *
9    * @author Oskari
10   */
11  public class TurnHistory {
12  
13      private final LinkedList<Turn> history;
14  
15      public TurnHistory() {
16          this.history = new LinkedList<>();
17      }
18  
19      public void addTurn(Turn turn) {
20          this.history.add(turn);
21      }
22  
23      public LinkedList<Turn> getList() {
24          return this.history;
25      }
26  
27      public void removeAfter(int vuoro) {
28          while (history.size() > vuoro) {
29              history.remove(vuoro - 1);
30          }
31      }
32  
33      public Turn getViimeinenVuoro() {
34          return history.getLast();
35      }
36  
37      public int getVuoroNumero() {
38          if (history.size() == 0) {
39              return 1;
40          }
41          if (history.getLast().isComplete()) {
42              return history.getLast().getTurnNumber() + 1;
43          }
44          return history.getLast().getTurnNumber();
45      }
46  
47      public static TurnHistory parseString(String string) {
48          TurnHistory retHistory = new TurnHistory();
49          String[] strings = string.split(Character.toString('\n'));
50          for (int i = 0; i < strings.length; i++) {
51              Turn turn = new Turn(strings[i]);
52              retHistory.addTurn(turn);
53          }
54          return retHistory;
55      }
56  
57      @Override
58      public String toString() {
59          String string = "";
60          for (Turn turn : history) {
61              string = string.concat(turn.toString());
62              if (turn.getTurnNumber() != history.getLast().getTurnNumber()) {
63                  string = string.concat("\n");
64              }
65          }
66          return string;
67      }
68  }
